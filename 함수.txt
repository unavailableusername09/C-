2.두 수 입력받아 중간값 출력하기

#include <stdio.h>
//두값의 중간값을 구하기

int mid(int a, int b); //함수 선엄

int main() {
	int n1,n2;
	scanf("%d %d", &n1,&n2); //숫자 입력받기
	printf("%d와(과) %d사이의 값은 %d입니다", n1,n2,mid(n1, n2));  
}

int mid(int a, int b) {  //함수 정의
	int result;  //중간값을 리턴하기위한 변수
	result = (a + b) / 2;
	return result;
}

3.세 수의 평균 구하기
#include <stdio.h>

double avg();

int main() {
	int a, b,c;
	scanf("%d %d %d", &a, &b,&c);

	printf("%g", avg(a, b,c));
}

double avg(int a, int b,int c) {
	return (a+ b + c) / 3.0;
}
