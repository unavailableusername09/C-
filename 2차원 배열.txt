5. 3행 3열 배열 2개를 만들어 입력받기

#include <stdio.h>

int main() {
	int a[3][3]={0,};
	int b[3][3] = { 0, };
	int i, j;
	for (i = 0; i < 3; i++) {
		printf("1번 배열 %d행: ", i + 1);
		for (j = 0; j < 3; j++) {
			scanf("%d", &a[i][j]);
		}
	}
	for (i = 0; i < 3; i++) {
		printf("2번 배열 %d행: ", i + 1);
		for (j = 0; j < 3; j++) {
			scanf("%d", &b[i][j]);
		}
	}
}

6. 3행 3열 배열 2개를 만들어 입력받고, 두 배열의 합 출력

#include <stdio.h>

int main() {
	int a[3][3]={0,};
	int b[3][3] = { 0, };
	int i, j;
	for (i = 0; i < 3; i++) {
		printf("1번 배열 %d행: ", i + 1);
		for (j = 0; j < 3; j++) {
			scanf("%d", &a[i][j]);
		}
	}
	for (i = 0; i < 3; i++) {
		printf("2번 배열 %d행: ", i + 1);
		for (j = 0; j < 3; j++) {
			scanf("%d", &b[i][j]);
		}
	}
	printf("=============================\n");

	for (i = 0; i < 3; i++) {
		for (j = 0; j < 3; j++)
			printf("%d ", a[i][j] + b[i][j]);
		printf("\n");
	}
}

7. 대문자열 3행 5열 입력받아 소문자로 출력

#include <stdio.h>

int main() {
	char a[3][5];
	int i, j;
	for (i = 0; i < 3; i++) {   
		for (j = 0; j < 5; j++){  
			scanf(" %c", &a[i][j]); 
		}
	}
	printf("=========================\n");
	for (i = 0; i < 3; i++) {
		for (j = 0; j < 5; j++)
			printf("%c ", a[i][j]+32);
		printf("\n");
	}
}

8.엔터칠때까지 입력받고 출력

#include <stdio.h>


char a[100000]={0,};

int main() {

	int i=0,j=0;
	while(1) {
		scanf("%c", &a[i++]);
		if (a[i - 1] == 10) break;
		j++;
	}
	for (i=0;i<=j;i++)
		printf("%c", a[i]);
}
