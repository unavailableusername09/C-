4.정수 입력받다가 개수 출력
#include <stdio.h>
int main() {
	int num, i;
	int a[11]={0};

	while (1) {
		scanf("%d", &num);
		if (num > 10 || num < 1)
			break;
		a[num]++;
	}
	for (i = 1; i <= 10; i++)
		if (a[i] > 0)
			printf("%d: %d개\n", i, a[i]);
}
