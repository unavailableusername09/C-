1.근무 연수
#include <stdio.h>
int main() {
	int years, bonus;
	printf("근무 연수를 입력하세요 ");
	scanf("%d", &years);
	if (years > 10) {
		bonus = 500;
		printf("근무연수가 %d년이므로, 보너스는 %d만원입니다.",years,bonus);
	}
	else {
		bonus = 300;
		printf("근무연수가 %d년이므로, 보너스는 %d만원입니다.",years,bonus);
	}
	return 0;
}

1.입력한 숫자 출력
#include <stdio.h>
int main() {
	int a;
	printf("숫자 입력(1~3): ");
	scanf("%d", &a);
	switch (a)
	{
	case 1:
	case 2:
	case 3:
		printf("%d번 입력",a);
		break;

	default:
		printf("다른 번호를 입력하였습니다");
		break;
	}
}
2. 선택적 사칙연산
#include <stdio.h>
int main() {
	int a,b,c;
	printf("정수 두개 입력: ");
	scanf("%d %d", &a,&b);
	printf("덧셈=1 뺄셈=2 곱셈=3 나눗셈=4\n");
	scanf("%d", &c);
	switch (c)
	{
	case 1:
		printf("%d",a+b);
		break;
	case 2:
		printf("%d",a-b);
		break;
	case 3:
		printf("%d",a*b);
		break;
	case 4:
		printf("%g", (float)a / b);
		break;
	default:
		printf("그런거 없다");
	}
}
