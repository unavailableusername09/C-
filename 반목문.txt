#include <stdio.h>

int main() {
	int a,b=0,c;
	printf("정수를 입력하시오: ");
	scanf("%d", &a);
	c = a;
	while (a > 0) {
		b += a;
		a--;
	}
	printf("1부터 %d까지의 합은 %d입니다", c, b);
}

int main() {
	int a=100,sum=0;
	while (a != 0) {
		printf("값을 입력하시오: ");
		scanf("%d", &a);
		sum += a;
	}
	if (sum != 0)
		printf("총 합은 %d입니다.", sum);
}
