5. 피타고라스 어쩌구 저쩌구
#include <stdio.h>
int main() {
	int a=1,b=1,c=1, sum = 0;
	for (; a <= 100; a++)
		for (b=1; b <= 100; b++)
			for (c=1; c <= 100; c++) {
				if (a * a + b * b == c * c && a>b) {
					printf("%d %d %d\n", a, b, c);
					sum++;
				}
			}
	printf("%d개",sum);
}

4.11(월)
1.자연수n을 입력받아 n줄만큼 출력하기(직각삼각형)

#include <stdio.h>
int main() {
	int n, a=1;
	scanf("%d", &n);
	for (int i = 0; i <= n; i++) {
		for (int j = 0; j < i; j++)
			printf("%-2d ", a++);
		printf("\n");
	}
}

2.공백이 있는 직각삼각형 출력하기
#include <stdio.h>
int main() {
	for (int i = 0; i < 5; i++) {
		for (int k =1; k < 5-i; k++)
			printf(" ");
		for (int j = 0; j <= i; j++)
			printf("*");
		printf("\n");
	}
3. 직각삼각형 출력
#include <stdio.h>
int main() {
	for (int i = 0; i < 5; i++) {
		for (int k = 0; k < i; k++)
			printf(" ");
		for (int j = 5; j > i; j--)
			printf("*");
		printf("\n");
	}
}

       for (i = 1; i <= j; i++) {

        }
